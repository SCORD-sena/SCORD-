/* ENCRIPTAR CONTRASEÑAS QUE ESTEN EN LA BASE DE DATOS YA ENCRIPTADAS */
UPDATE personas
SET contraseña = EncriptarSHA2_512(contraseña)
WHERE LENGTH(contraseña) < 60;

/*MOSTRAR CORREO, CONTRASEÑA Y LA LONGITUD DE LA CONTRASEÑA*/
SELECT correo, contraseña, LENGTH(contraseña) AS longitud
FROM personas
ORDER BY longitud DESC;
